#4

SELECT client_id, COUNT(domain_name), DATE_FORMAT(created_datetime, '%M'), DATE_FORMAT(created_datetime, '%Y')
FROM sites
WHERE client_id = 1
GROUP BY domain_name
ORDER BY DATE_FORMAT(created_datetime, '%Y') ASC;

#5

SELECT sites.domain_name, COUNT(leads_id), DATE_FORMAT(leads.registered_datetime, '%M %d, %Y')
FROM sites
JOIN leads
ON sites.site_id = leads.site_id
WHERE DATE_FORMAT(registered_datetime, '%Y%m%d') BETWEEN  20110101 AND 20110215
GROUP BY leads.registered_datetime
ORDER BY leads_id

SELECT concat(clients.first_name, ' ', clients.last_name), COUNT(leads.leads_id)
FROM clients
JOIN sites ON sites.client_id = clients.client_id
JOIN leads ON sites.site_id = leads.site_id
WHERE DATE_FORMAT(registered_datetime, '%Y%m%d') BETWEEN  20110101 AND 20111231
GROUP BY concat(clients.first_name, ' ', clients.last_name);

FROM clients
JOIN sites ON sites.client_id = clients.client_id

PART II

SELECT concat(clients.first_name, ' ', clients.last_name), SUM(billing.amount), DATE_FORMAT(billing.charged_datetime, '%M'), DATE_FORMAT(billing.charged_datetime, '%Y')
FROM clients
JOIN billing
ON clients.client_id = billing.client_id
GROUP BY billing.charged_datetime
ORDER BY clients.client_id, DATE_FORMAT(billing.charged_datetime, '%Y'), DATE_FORMAT(billing.charged_datetime, '%m')

SELECT concat(clients.first_name, ' ', clients.last_name), GROUP_CONCAT(domain_name SEPARATOR ' / ') AS Websites
FROM clients
JOIN sites
ON clients.client_id = sites.client_id
GROUP BY concat(clients.first_name, ' ', clients.last_name)
ORDER BY clients.client_id
